                 var json = File.ReadAllText(simetraConfigPath);
                 var jsonOptions = new System.Text.Json.JsonSerializerOptions

                     ReadCommentHandling = System.Text.Json.JsonCommentHandling.Skip,

                     PropertyNameCaseInsensitive = true
